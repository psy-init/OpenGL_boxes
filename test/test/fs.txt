#version 330 core

in vec2 TexCoord;
out vec4 FragColor;

uniform float color;
uniform sampler2D texture1;
uniform sampler2D texture2;

void main(){
	FragColor = mix(texture(texture1, TexCoord), texture(texture2, TexCoord), color) * vec4(TexCoord, color, 1.0f);
	//FragColor = texture(texture2, TexCoord);
}
